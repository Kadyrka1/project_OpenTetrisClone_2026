/*
Solid White - GND (Ground)
Short Dash - - - - - Right
Triangle ▲▲▲▲ - Up
"X" Mark XXXX - Down
Long Dash _____ -  Left
*/

// ===== JOYSTICK PIN DEFINITIONS =====
#define PIN_UP     18
#define PIN_DOWN   19
#define PIN_LEFT   21
#define PIN_RIGHT  22

unsigned long lastPrint = 0;
const unsigned long interval = 500; // 0.5 seconds

void setup() {
  Serial.begin(115200);

  pinMode(PIN_UP, INPUT_PULLUP);
  pinMode(PIN_DOWN, INPUT_PULLUP);
  pinMode(PIN_LEFT, INPUT_PULLUP);
  pinMode(PIN_RIGHT, INPUT_PULLUP);

  Serial.println("Joystick test started");
  Serial.println("UP | DOWN | LEFT | RIGHT");
}

void loop() {
  if (millis() - lastPrint >= interval) {
    lastPrint = millis();

    bool up    = digitalRead(PIN_UP)    == LOW;
    bool down  = digitalRead(PIN_DOWN)  == LOW;
    bool left  = digitalRead(PIN_LEFT)  == LOW;
    bool right = digitalRead(PIN_RIGHT) == LOW;

    Serial.print("UP: ");
    Serial.print(up ? "PRESSED" : "----");

    Serial.print(" | DOWN: ");
    Serial.print(down ? "PRESSED" : "----");

    Serial.print(" | LEFT: ");
    Serial.print(left ? "PRESSED" : "----");

    Serial.print(" | RIGHT: ");
    Serial.println(right ? "PRESSED" : "----");
  }
}
